        @classmethod
        def validate_schedule_time(cls, v: Optional[str]) -> Optional[str]:
            if v is None:
                return v
            time_part, tz_part = v.rsplit(" ", 1)
            hh, mm = map(int, time_part.split(":"))
            if not (0 <= hh < 24) or not (0 <= mm < 60):
                raise ValueError(f"schedule_time hour/minute out of range: {v}")
            try:
                ZoneInfo(tz_part)  # Confirms IANA tz exists (tzdata dep per Pitfall 5)
            except ZoneInfoNotFoundError:

            return v
