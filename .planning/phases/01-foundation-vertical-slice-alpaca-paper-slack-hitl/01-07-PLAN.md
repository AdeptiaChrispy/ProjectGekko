        from dataclasses import dataclass, field
        import time
        from gekko.core.errors import BudgetExceeded
        from gekko.logging_config import get_logger

        log = get_logger(__name__)

        @dataclass
        class BudgetTracker:
            soft_max_calls: int = 12
            soft_max_tokens: int = 8000
            soft_max_seconds: float = 60.0
            started_at: float = field(default_factory=time.monotonic)
            calls: int = 0
            tokens_used: int = 0

            def record_call(self, tokens: int) -> None:
                self.calls += 1
                self.tokens_used += tokens
                elapsed = time.monotonic() - self.started_at
                if self.calls > self.soft_max_calls or self.tokens_used > self.soft_max_tokens or elapsed > self.soft_max_seconds:
                    log.warning(
                        "research.budget.soft_exceeded",
                        calls=self.calls, tokens=self.tokens_used, elapsed=elapsed,
                    )
                if self.calls > 2 * self.soft_max_calls or self.tokens_used > 2 * self.soft_max_tokens or elapsed > 2 * self.soft_max_seconds:
                    raise BudgetExceeded(f"per-cycle budget 2x exceeded: calls={self.calls}, tokens={self.tokens_used}, seconds={elapsed:.1f}")

            def to_dict(self) -> dict[str, Any]:
                return {"calls": self.calls, "tokens": self.tokens_used, "seconds": time.monotonic() - self.started_at}

        async def write_proposal(
            session,
            *,
            user_id: str,
            strategy: Strategy,
            strategy_db_id: str,
            run_id: str,
            decision_id: str,
            tool_outcome: str,   # "propose_trade" or "propose_no_action"
            payload: dict,        # raw kwargs from the LLM tool call
            prompt_model: str = "claude-sonnet-4-6",
        ) -> TradeProposal | NoActionProposal:
            now = datetime.now(timezone.utc).isoformat()

            if tool_outcome == "propose_trade":
                # Validate; LLM-supplied dict -> Pydantic
                tp = TradeProposal.model_validate({
                    **payload,
                    "user_id": user_id,
                    "strategy_name": strategy.name,
                    "decision_id": decision_id,
                    "client_order_id": "x" * 32,  # placeholder; computed below
                })
                # Watchlist enforcement (P1 hallucinated-ticker guard)
                if tp.ticker.upper() not in strategy.watchlist:
                    await append_event(
                        session, user_id=user_id, strategy_id=strategy_db_id,
                        event_type="error",
                        payload={
                            "context": "proposal_writer.watchlist_violation",
                            "error_class": "ProposalRejected",
                            "error_message": f"Hallucinated ticker {tp.ticker} not in watchlist {strategy.watchlist}",
                            "rejected_proposal": payload,
                        },
                    )
                    raise ProposalRejected(f"Ticker {tp.ticker} not in strategy watchlist")

                # Compute deterministic client_order_id (D-20)
                client_order_id = compute_client_order_id(
                    strategy_id=strategy_db_id,
                    decision_id=decision_id,
                    side=tp.side,
                    qty=tp.qty,
                    ticker=tp.ticker,
                )
                tp = tp.model_copy(update={"client_order_id": client_order_id})

                # Idempotent persistence — check for existing row with same decision_id
                from sqlalchemy import select
                existing = (await session.execute(select(ProposalRow).where(ProposalRow.proposal_id == decision_id))).scalar_one_or_none()
                if existing:
                    return tp  # Already written; idempotent return

                # Insert proposal row
                session.add(ProposalRow(
                    proposal_id=decision_id,  # decision_id IS the proposal_id (1:1)
                    user_id=user_id,
                    strategy_id=strategy_db_id,
                    status="PENDING",
                    payload_json=tp.model_dump_json(),
                    client_order_id=client_order_id,
                    created_at=now,
                    updated_at=now,
                ))
                await session.flush()

                # Audit: decision event (the "we ran the decision agent" record per D-15)
                decision_payload = normalize_decimals({
                    "run_id": run_id,
                    "strategy_id": strategy_db_id,
                    "prompt_model": prompt_model,
                    "research_brief_run_id": run_id,
                    "decision_outcome": "trade",
                })
                await append_event(session, user_id=user_id, strategy_id=strategy_db_id, event_type="decision", payload=decision_payload)

                # Audit: proposal event with FULL structured rationale (D-15)
                proposal_payload = normalize_decimals(tp.model_dump(mode="json"))
                await append_event(session, user_id=user_id, strategy_id=strategy_db_id, event_type="proposal", payload=proposal_payload)

                return tp

            elif tool_outcome == "propose_no_action":

                    **payload, "user_id": user_id, "strategy_name": strategy.name, "decision_id": decision_id,

                    "run_id": run_id, "strategy_id": strategy_db_id, "prompt_model": prompt_model,
                    "research_brief_run_id": run_id, "decision_outcome": "no_action",
                }))
                await append_event(session, user_id=user_id, strategy_id=strategy_db_id, event_type="proposal", payload=normalize_decimals(nap.model_dump(mode="json")))
                return nap

            else:
                raise ValueError(f"Unknown tool_outcome: {tool_outcome}")
