        from datetime import datetime, timezone
        from zoneinfo import ZoneInfo
        import pandas_market_calendars as mcal
        from functools import lru_cache

        NYSE_TZ = ZoneInfo("America/New_York")

        @lru_cache(maxsize=1)
        def _nyse_calendar():
            return mcal.get_calendar("NYSE")

        def is_market_open(now: datetime | None = None) -> bool:
            """True if NYSE is currently in RTH (09:30-16:00 ET, weekdays, non-holidays, accounting for half-days)."""
            if now is None:
                now = datetime.now(timezone.utc)
            if now.tzinfo is None:
                now = now.replace(tzinfo=timezone.utc)
            cal = _nyse_calendar()
            # Use schedule() to get session start/end for the date
            local_date = now.astimezone(NYSE_TZ).date()
            sched = cal.schedule(start_date=local_date, end_date=local_date)
            if sched.empty:
                return False  # weekend/holiday/closed
            # sched.iloc[0] has market_open and market_close as pandas Timestamps in UTC
            open_ts = sched.iloc[0]["market_open"].to_pydatetime()
            close_ts = sched.iloc[0]["market_close"].to_pydatetime()
            return open_ts <= now < close_ts

        def next_market_open(now: datetime | None = None) -> datetime:
            """Next NYSE session open after `now` (UTC datetime)."""
            if now is None:
                now = datetime.now(timezone.utc)
            cal = _nyse_calendar()
            local_date = now.astimezone(NYSE_TZ).date()
            # Look up to 10 days ahead (covers long weekends + holiday clusters)
            sched = cal.schedule(start_date=local_date, end_date=local_date.replace(day=local_date.day) + timedelta(days=10))
            for _, row in sched.iterrows():
                open_ts = row["market_open"].to_pydatetime()
                if open_ts > now:
                    return open_ts
            raise RuntimeError("No upcoming NYSE session found within 10 days — calendar data may be missing")

            ("PENDING", "APPROVED"),
            ("PENDING", "REJECTED"),
            ("APPROVED", "EXECUTING"),
            ("EXECUTING", "FILLED"),
            ("EXECUTING", "FAILED"),

        async def transition_status(session, proposal_id: str, *, from_status: str, to_status: str) -> Proposal | None:
            """Atomic PENDING->APPROVED state transition. Returns the row, or None if no-op (idempotent for same-state)."""
            row = (await session.execute(select(Proposal).where(Proposal.proposal_id == proposal_id))).scalar_one()
            if row.status == to_status:
                return row  # idempotent
            if (row.status, to_status) not in STATE_TRANSITIONS:
                raise ValueError(f"Invalid status transition: {row.status} -> {to_status}")
            row.status = to_status
            row.updated_at = datetime.now(timezone.utc).isoformat()
            await session.flush()
            return row
