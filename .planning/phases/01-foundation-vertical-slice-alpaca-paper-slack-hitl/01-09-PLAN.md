        import getpass
        from typing import Optional

        _passphrase: Optional[str] = None

        def prompt_passphrase(prompt_text: str = "SQLCipher passphrase: ") -> str:
            """Read passphrase from stdin (interactive) and cache for the process lifetime.
            D-19: never persisted, never logged. Lives in process memory only."""
            global _passphrase
            if _passphrase is not None:
                return _passphrase
            _passphrase = getpass.getpass(prompt_text)
            return _passphrase

        def set_passphrase(passphrase: str) -> None:
            """For non-interactive contexts (env var or test fixture) — sets the cache directly."""
            global _passphrase
            _passphrase = passphrase

        def get_passphrase() -> str:
            """Returns the cached passphrase; raises if not set."""
            if _passphrase is None:
                raise RuntimeError("Passphrase not set. Run `gekko init` or call prompt_passphrase() at process start.")
            return _passphrase

        from apscheduler.schedulers.asyncio import AsyncIOScheduler
        from apscheduler.jobstores.sqlalchemy import SQLAlchemyJobStore
        from apscheduler.triggers.cron import CronTrigger
        from zoneinfo import ZoneInfo

        def build_scheduler(db_url: str) -> AsyncIOScheduler:
            """Build a single AsyncIOScheduler with SQLAlchemyJobStore on the app SQLite (SQLCipher) DB.

            NOTE: APScheduler's SQLAlchemyJobStore uses synchronous SQLAlchemy under the hood. We pass
            the same db_url as the rest of the app but constructed for sync access. Per RESEARCH §A7,
            single SQLCipher DB hosts both app data and job store without contention (SQLite WAL).
            """
            jobstores = {"default": SQLAlchemyJobStore(url=db_url)}
            return AsyncIOScheduler(jobstores=jobstores, timezone="UTC")

        def schedule_strategy_daily(
            scheduler: AsyncIOScheduler, *, user_id: str, strategy_name: str, schedule_time: str,
        ) -> str:
            """schedule_time format: 'HH:MM IANA_TZ' e.g. '10:00 America/New_York' (D-08)."""
            time_part, tz_part = schedule_time.rsplit(" ", 1)
            hh, mm = map(int, time_part.split(":"))
            try:
                tz = ZoneInfo(tz_part)  # requires tzdata on Windows per Pitfall 5

                args=[user_id, strategy_name],
                kwargs={"source": "schedule"},
                id=job_id,
                replace_existing=True,
            )
            return job_id

        def unschedule_strategy(scheduler, *, user_id: str, strategy_name: str) -> bool:
            job_id = f"run-{user_id}-{strategy_name}"
            try:
                scheduler.remove_job(job_id)
